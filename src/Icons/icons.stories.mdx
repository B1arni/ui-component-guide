import { Meta, Story, Canvas } from "@storybook/addon-docs/blocks";

<Meta title="Icon" component={<i />} />

export const Template = (args) => <i {...args} />;

## Icon Props

Icons can be used using `i` tag. See the examples.

| Name      | Type            | default | Description                                                                                                    |
| --------- | --------------- | ------- | -------------------------------------------------------------------------------------------------------------- |
| className | text            |         | accepts the class of expected icon. For full list of icons visit [docs](https://icons.getbootstrap.com/#icons) |
| style     | `CSSProperties` |         | list of css properties as object                                                                               |

<Canvas withSource withToolbar>
